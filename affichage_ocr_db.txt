-- Table des utilisateurs
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(50) DEFAULT 'correcteur',
    is_active BOOLEAN DEFAULT true
);

-- Table des documents (fichiers uploadés)
CREATE TABLE documents (
    id SERIAL PRIMARY KEY,
    filename VARCHAR(255) NOT NULL,
    uploaded_by VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Table des données d'affichage (résultats OCR)
CREATE TABLE affichage_data (
    id SERIAL PRIMARY KEY,
    document_id INTEGER REFERENCES documents(id) ON DELETE CASCADE,
    numero_affichage VARCHAR(100),
    lettre_date VARCHAR(50),
    requerant VARCHAR(255),
    parcelle VARCHAR(100),
    section VARCHAR(100),
    commune VARCHAR(100),
    lieu_dit VARCHAR(255),
    extraction_ocr TEXT,
    statut VARCHAR(50) DEFAULT 'en attente',
    valide_par VARCHAR(100),
    validated_at TIMESTAMP,
    upload_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
