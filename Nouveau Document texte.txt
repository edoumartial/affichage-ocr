-- Création de la base de données
CREATE DATABASE affichage_ocr_db;

-- Se connecter à la base de données (si vous utilisez psql, utilisez \c affichage_ocr_db)

-- 1. Création de la table 'documents'
CREATE TABLE documents (
    id SERIAL PRIMARY KEY,
    filename CHARACTER VARYING(255),
    filepath CHARACTER VARYING(512),
    raw_text TEXT,
    status CHARACTER VARYING(50),
    uploaded_by CHARACTER VARYING(255),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. Création de la table 'users'
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username CHARACTER VARYING(50) UNIQUE,
    password_hash CHARACTER VARYING(255),
    role CHARACTER VARYING(20),
    is_active BOOLEAN DEFAULT TRUE
);

-- 3. Création de la table 'affichage_data' (adaptée selon vos besoins)
CREATE TABLE affichage_data (
    id SERIAL PRIMARY KEY,
    document_id INTEGER REFERENCES documents(id),
    
    -- Colonnes personnalisées
    lettre_date CHARACTER VARYING(100),
    requerant TEXT,
    parcelle TEXT,
    section TEXT,
    commune TEXT,
    lieu_dit TEXT,
    
    -- Colonnes héritées conservées (adaptées au contexte)
    extraction_ocr TEXT,
    statut CHARACTER VARYING(20),
    valide_par CHARACTER VARYING(100),
    validated_at TIMESTAMP WITHOUT TIME ZONE,
    
    -- Clé étrangère vers l'utilisateur (le validateur)
    utilisateur CHARACTER VARYING(255) REFERENCES users(username)
);
